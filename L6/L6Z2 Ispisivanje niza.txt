#include <stdio.h>

int ispisi(int* pok, int vel) {
  int i;
  printf("\n");
  printf("{");
  for(i=0;i<vel;i++){
      printf("%d", *pok);
      pok++;
      if(i<vel-1){
          printf(", ");
      }
      else{
          printf("}");
      }
  }
  if(vel==0){
      printf("}");
  }
  return *pok;
}
/* Ovdje napisite funkciju ispisi */

  int niz2[10] = {0, 3, 2, 2, 2, 4, 6, 1, 9, 10};

  ispisi(niz2, 10);
  return 0;
}
